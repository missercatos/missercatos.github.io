//总结： 遍历 暴力 容器
#include <iostream>
#include <vector>
using namespace std;

/*
洛谷P1003 铺地毯 解题思路：
1. 问题描述：在平面上依次铺放n块地毯，每块地毯用左下角坐标(a,b)、长度g、宽度k定义。
   给出一个查询点(x,y)，需要找出这个点被最上面的哪块地毯覆盖（即最后铺放的地毯在最上面）。
   如果没有地毯覆盖该点，则输出-1。
2. 关键点：由于后铺的地毯覆盖先铺的地毯，因此需要从最后一块地毯开始逆序检查。
   对于每块地毯，检查点(x,y)是否在其范围内：a <= x <= a+g 且 b <= y <= b+k。
3. 算法步骤：
   - 读取n，然后读取n行地毯信息，存储到数组中。
   - 读取查询点(x,y)。
   - 从i=n-1到0逆序循环，检查点是否在当前地毯范围内。
   - 如果找到，输出i+1（地毯编号从1开始）并返回。
   - 如果循环结束未找到，输出-1。
4. 时间复杂度：O(n)，空间复杂度：O(n)。
*/

int main() {
    int n;
    cin >> n;
    
    // 使用vector存储地毯信息，每个地毯是一个包含四个整数的vector
    vector<vector<int>> carpets(n, vector<int>(4));
    
    for (int i = 0; i < n; i++) {
        // 读取地毯的左下角坐标(a,b)和长度g、宽度k
        cin >> carpets[i][0] >> carpets[i][1] >> carpets[i][2] >> carpets[i][3];
    }
    
    int x, y;
    cin >> x >> y;
    
    // 从最后一块地毯开始逆序检查（最后铺的地毯在最上面）
    for (int i = n - 1; i >= 0; i--) {
        int a = carpets[i][0];
        int b = carpets[i][1];
        int g = carpets[i][2];
        int k = carpets[i][3];
        
        // 检查点(x,y)是否在当前地毯范围内
        // 注意：地毯覆盖区域为 [a, a+g] 和 [b, b+k]
        if (x >= a && x <= a + g && y >= b && y <= b + k) {
            // 找到覆盖点的最上面的地毯，输出编号（从1开始）
            cout << i + 1 << endl;
            return 0;
        }
    }
  
    // 没有地毯覆盖该点
    cout << -1 << endl;
    return 0; 
    //这里把else的输出结果放在外面是因为，如果在里面，如果满足if的条件或是满足else的条件就直接输出了，不能完全遍历。
    //如果在外面，当有不满足if语句的时候，会接着往下检查，知道满足if输出，如果全部没有满足就会退出，输出-1  
}
